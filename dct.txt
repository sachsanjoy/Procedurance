import pandas as pd
import seaborn as sns
import matplotlib.pyplot as plt
import numpy as np

# Generate example data
n = 100  # number of rows
data = {
    'a': np.random.rand(n),
    'b': np.random.rand(n)
}

# Create DataFrame
df = pd.DataFrame(data)

# Initialize the correlation matrix
corr_matrix = np.zeros((n, n))

# Compute the correlation between each pair of data points
for i in range(n):
    for j in range(n):
        corr_matrix[i, j] = np.corrcoef(df['a'][i], df['b'][j])[0, 1]

# Convert the correlation matrix to a DataFrame for plotting
corr_df = pd.DataFrame(corr_matrix)

# Create a heatmap using seaborn
plt.figure(figsize=(10, 8))
sns.heatmap(corr_df, annot=False, cmap='coolwarm', vmin=-1, vmax=1)
plt.title('Correlation Matrix between Data Points in Columns a and b')
plt.xlabel('b')
plt.ylabel('a')
plt.show()
