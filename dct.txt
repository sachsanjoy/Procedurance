import os
import fnmatch

def find_ipr_msg_files(directory):
    # List to store the found files
    ipr_msg_files = []
    
    # Traverse the directory tree
    for root, dirs, files in os.walk(directory):
        # Filter files that start with "IPR" and end with ".msg"
        for file in fnmatch.filter(files, 'IPR*.msg'):
            # Add the full path to the result list
            ipr_msg_files.append(os.path.join(root, file))
    
    return ipr_msg_files

# Example usage
directory_path = '/path/to/directory'
result = find_ipr_msg_files(directory_path)

# Print the list of found files
for file_path in result:
    print(file_path)
