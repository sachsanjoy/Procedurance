import pandas as pd
import seaborn as sns
import matplotlib.pyplot as plt
import numpy as np

# Generate example data
n = 100  # number of rows
data = {
    'a': np.random.rand(n),
    'b': np.random.rand(n)
}

# Create DataFrame
df = pd.DataFrame(data)

# Initialize the similarity matrix
similarity_matrix = np.zeros((n, n))

# Compute the similarity (absolute differences) between each pair of data points
for i in range(n):
    for j in range(n):
        similarity_matrix[i, j] = abs(df['a'][i] - df['b'][j])

# Convert the similarity matrix to a DataFrame for plotting
similarity_df = pd.DataFrame(similarity_matrix)

# Create a heatmap using seaborn
plt.figure(figsize=(10, 8))
sns.heatmap(similarity_df, annot=False, cmap='coolwarm', vmin=0, vmax=np.max(similarity_matrix))
plt.title('Similarity Matrix between Data Points in Columns a and b')
plt.xlabel('b')
plt.ylabel('a')
plt.show()
