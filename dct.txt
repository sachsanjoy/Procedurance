# Define the two lists
A = [1, 2, 3, 4, 5]
B = [4, 5, 6, 7, 8]

# Find elements in B that are missing in A
missing_in_A = [item for item in B if item not in A]

# Find elements in A that are missing in B
missing_in_B = [item for item in A if item not in B]

print(f"Elements missing in A compared to B: {missing_in_A}")
print(f"Elements missing in B compared to A: {missing_in_B}")
