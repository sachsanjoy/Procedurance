from flask import Flask
import dash
from dash import dcc, html
import plotly.graph_objs as go

# Initialize Flask
server = Flask(__name__)

# Initialize Dash
app = dash.Dash(__name__, server=server, url_base_pathname='/dashboard/')

# Sample Data
x_data = [1, 2, 3, 4, 5]
y_data = [10, 12, 8, 15, 10]

# Create Dash Layout
app.layout = html.Div(children=[
    html.H1(children='Simple Plotly Dashboard'),

    dcc.Graph(
        id='example-graph',
        figure={
            'data': [
                go.Scatter(
                    x=x_data,
                    y=y_data,
                    mode='lines+markers',
                    name='Line Plot'
                )
            ],
            'layout': go.Layout(
                title='Sample Line Plot',
                xaxis={'title': 'X-axis'},
                yaxis={'title': 'Y-axis'},
            )
        }
    )
])

# Define Flask route
@server.route('/')
def home():
    return 'Hello! Go to /dashboard/ to see the Plotly Dash dashboard.'

# Run the app
if __name__ == '__main__':
    server.run(debug=True)
