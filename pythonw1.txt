import re

input_text = 'IF Q8 GREATER THAN 4'
match = re.match(r'IF (\w+) GREATER THAN (\d+)', input_text)
if match:
    column_name = match.group(1)
    threshold = match.group(2)
    output_text = f"data[data['{column_name}']>{threshold}]"
    print(output_text)
else:
    print('Invalid input text')
